                }

                $.ajax({
                    dataType: "json",                                                                             
                    url: "/modules/search/services/search.ashx?Query=" + encodeURIComponent(request.term) + "&site=default_frontend&collection='default_collection'&start=0&NumberOfResults=" + start,

                    cache: false,
                    success: function (data) {                                                
                        datasource = [];
                        if (data != null) {
                            $("#search-wrapper").show();

                                datasource[i] = { label: data[i].Text, desc: data[i].ShortDescription, url: data[i].Url, filetype: data[i].FileType, empty: false };
                            }
                            response(datasource);
                        }
                        else
                            response([{ empty: true }]);
                    },
                    error: function (r) {
                        response([{ empty: true }]);
                    }
                });
